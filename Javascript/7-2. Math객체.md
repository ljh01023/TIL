# math

```javascript
// 오일러의 상수
console.log(Math.E); // 2.718281828459045

// 원주율
console.log(Math.PI); // 3.141592653589793

// 많이사용하는 내용
// 절대값
console.log(Math.abs(-10)); // 10

// 소수점 이하 올림
console.log(Math.ceil(1.4)); //2

// 소수점 이하 내림
console.log(Math.floor(1.4)); // 1

// 소수점 이하 반올림
console.log(Math.round(1.4)); // 1
console.log(Math.round(1.5)); // 2

// 정수만 반환
console.log(Math.trunc(1.5235)); // 1

// 최대값 반환
console.log(Math.max(1, 5, 8, 50)); // 50

// 최소값 반환
console.log(Math.min(1, 5, 8, 50)); // 1

// 거듭제곱 연산자 방법
console.log(3 ** 2); // 9

// 거듭제곱
console.log(Math.pow(3, 2)); // 9

// 0,1사이의 랜덤 숫자 반환 (난수발생)
console.log(Math.random()); // 0.8162764616199754

// 소숫자리 제거 후 정수 반환
console.log(Math.trunc(Math.random() * 100)); // 75
```
